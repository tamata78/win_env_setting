```

# apply name, email to settings
```
$ pwd
/c/Users/username/work

$ git config --global user.name "TaroYamada"
$ git config --global user.email "taro.yamada@email.com"

$ git config -l
user.name=TaroYamada
user.email=taro.yamada@email.com
```

# Github ssh setting
```
$ ssh-keygen -t rsa -b 4096 -C '5101956+tamata78@users.noreply.github.com'

# copy pub_key to clipboad, set pub_key to github ssh key setting
$ clip < ~/.ssh/id_rsa.pub

# check connect with github
$ ssh -T git@github.com
```

# dotfiles setting
```
touch ~/.bash_profile

source ~/.bash_profile # if it doesnot start restart
